 16class Idable:
 17    def __init__(self, ids:list=None, mainidc:str=None, id=None, code:str=None):
 18        self.ids = ids
 19        if self.ids is None:
 20            self.ids = []
 21        self.mainidc = mainidc
 22        if id is not None:
 23            if ids is not None:
 24                raise Exception("Idable doesn't accept a list of ids if a single id is passed.")
 25            if mainidc is None:
 26                self.mainidc = code
 27            self.ids.append(Identifier(id=id, code=code))
 28    def identifier(self, code:str=None) -> Identifier:
 29        if code == None:
 30            code = self.mainidc
 31        for identifier in self.ids:
 32            if identifier.code == code:
 33                return identifier
 34    def id(self, code:str=None) -> str:
 35        if self.identifier(code) is not None:
 36            return self.identifier(code).id
 37        return None
 38    def set_id(self, code:str=None, id:str=None):
 39        for identifier in self.ids:
 40            if identifier.code == code:
 41                identifier.id = id
 42                return
 43        self.ids.append( Idable(id=id, code=code) )
 44    def iddict(self, *idcs):
 45        d = self.__dict__.copy()
 46        if idcs == None or len(idcs) == 0:
 47            idcs = [id.code for id in self.ids]
 48        for idc in idcs:
 49            d[idc] = self.id(idc)
 50        del d["ids"]
 51        del d["mainidc"]
 52        return d
 53    def __str__(self):
 54        if self.id() is not None:
 55            return self.id()
 56        return ""
